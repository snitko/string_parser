  def urls_to_images(options = {})
    wrap_with = options[:wrap_with] || ['','']
    @modified_string.gsub!(
      /(\s|^|\A|\n|\t|\r)(http:\/\/.*?\.(jpg|jpeg|png|gif|JPG|JPEG|PNG|GIF))([,.])?(\s|$|\n|\Z|\t|\r)/,
      "#{wrap_with[0]}<img src=\"\\2\" alt=\"\"/>\\5#{wrap_with[1]}"
    )
    self
  end
