  def break_long_words(length=75)
    @modified_string.gsub!(/(href="|src=")?[^\s^\n^\^^\A^\t^\r]{#{length},}/) do |s|
      unless $1
        last_pos = 0; result_string = ''
          while string = s[last_pos..(last_pos + 75)]
          result_string += string + ' '
          last_pos += 75
        end
        result_string
      else
        s
      end
    end
    self
  end
